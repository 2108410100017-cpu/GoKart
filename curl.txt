#!/bin/bash
# Airflow Setup Script for WSL Ubuntu
# Python 3.11 + Airflow 2.7.2
# Run: bash setup_airflow.sh

# 1Ô∏è‚É£ Update and install prerequisites
sudo apt update && sudo apt upgrade -y
sudo apt install -y software-properties-common curl git

# 2Ô∏è‚É£ Add deadsnakes PPA for Python 3.11
sudo add-apt-repository -y ppa:deadsnakes/ppa
sudo apt update
sudo apt install -y python3.11 python3.11-venv python3.11-distutils python3.11-dev build-essential

# 3Ô∏è‚É£ Create project directory in WSL home
PROJECT_DIR=~/goKart
mkdir -p $PROJECT_DIR
cd $PROJECT_DIR

# 4Ô∏è‚É£ Create Python 3.11 virtual environment
python3.11 -m venv airflow_venv311

# 5Ô∏è‚É£ Activate virtual environment
source airflow_venv311/bin/activate

# 6Ô∏è‚É£ Upgrade pip, setuptools, wheel
python -m pip install --upgrade pip setuptools wheel

# 7Ô∏è‚É£ Install Airflow with proper constraints
AIRFLOW_VERSION=2.7.2
PYTHON_VERSION=3.11
CONSTRAINT_URL="https://raw.githubusercontent.com/apache/airflow/constraints-${AIRFLOW_VERSION}/constraints-${PYTHON_VERSION}.txt"

pip install "apache-airflow==${AIRFLOW_VERSION}" --constraint "${CONSTRAINT_URL}"

# 8Ô∏è‚É£ Set AIRFLOW_HOME
export AIRFLOW_HOME=$PROJECT_DIR/airflow
mkdir -p $AIRFLOW_HOME

# 9Ô∏è‚É£ Initialize Airflow database
airflow db init

# üîü Create admin user
airflow users create \
    --username admin \
    --password admin \
    --firstname Admin \
    --lastname User \
    --role Admin \
    --email admin@example.com

echo "‚úÖ Airflow setup complete!"
echo "To run Airflow Scheduler: source airflow_venv311/bin/activate && airflow scheduler"
echo "To run Airflow Webserver: source airflow_venv311/bin/activate && airflow webserver --port 8080"
echo "Airflow UI: http://localhost:8080"
